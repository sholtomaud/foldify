#!/bin/bash

blue=$(tput setaf 4)
normal=$(tput sgr0)

printf  "%40s\n" "Hello ${blue}TWJH${normal}"
printf  "This script will download the following" 
printf  "* Nodejs Verson Manager to enable to you control nodejs versions"
printf  "* A recent version of nodejs and the Node Package Manager npm \n\n"

printf  "Using homebrew to install nvm"
brew install nvm

printf  "\nMaking an nvm folder in user directory"
mkdir ~/.nvm

printf  "\nInstalling latest stable version of nodejs and npm so we get the npx utility"
/usr/local/opt/nvm/nvm.sh install stable

brew cask install basictex
sudo tlmgr install calculator circuitikz amsmath mathtools amssymb booktabs tabularx bondgraphs pgfplots bm listings multirow graphicx tikz textcomp color adjustbox
